# Ensure RDP is enabled (belt and braces)
- name: Enable RDP
  win_regedit:
    path: HKLM:\System\CurrentControlSet\Control\Terminal Server
    name: fDenyTSConnections
    data: 0
    type: dword

- name: Enable firewall rule for RDP
  win_firewall_rule:
    name: Remote Desktop
    enable: yes

# Allow domain RDP group
- name: Add RDP-Access group to local RDP users
  win_group_membership:
    name: "Remote Desktop Users"
    members:
      - "JLBCredit\\RDP-Access"
    state: present

# Optional but highly recommended: install RSAT tools
- name: Install RSAT tools
  win_feature:
    name:
      - RSAT-AD-Tools
      - RSAT-DNS-Server
      - RSAT-Group-Policy
    state: present
    include_management_tools: yes
