# roles/common/tasks/main.yml
- name: Enable RDP
  win_regedit:
    path: HKLM:\System\CurrentControlSet\Control\Terminal Server
    name: fDenyTSConnections
    data: 0
    type: dword

- name: Enable firewall rule for RDP
  win_firewall_rule:
    name: Remote Desktop
    enable: yes

- name: Set DNS server
  win_dns_client:
    adapter_names: "*"
    ipv4_addresses:
      - "{{ dns_server }}"

- name: Sync time with domain
  win_service:
    name: w32time
    start_mode: auto
    state: started

- name: Force time resync
  win_shell: w32tm /resync
